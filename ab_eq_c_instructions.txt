# This Turing machine accepts the language

# {a^m b^n c^m+n : m ≥ 0, n ≥ 0}


input_alphabet = {a, b, c}

tape_alphabet = {a, b, c, _}

# qaccept (@) and qreject (!) are implied
states = {q0, qr, qR, ql}


q0 a -> qr _ R

q0 b -> qr _ R

q0 c -> q!

q0 _ -> q@


qr a -> qr a R

qr b -> qr b R

qr c -> qr c R

qr _ -> qR _ L


qR a -> q!

qR b -> q!

qR c -> ql _ L

qR _ -> q!


ql a -> ql a L

ql b -> ql b L

ql c -> ql c L

ql _ -> q0 _ R
