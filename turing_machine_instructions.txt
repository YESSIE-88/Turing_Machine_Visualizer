# {a^m b^n c^m+n : m ≥ 0, n ≥ 0}

Gamma = {a, b, c, _} # Tape Alphabet


q0 a -> qr _ R

q0 b -> qr _ R

q0 c -> qreject

q0 _ -> qaccept

qr b -> qr b R

qr c -> qr c R

qr _ -> qR _ L

qR c -> ql _ L

ql b -> ql b L

ql c -> ql c L
